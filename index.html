    <script>

        // 语音合成

        let synth = window.speechSynthesis;

        let utterance = null;

        let isPlaying = false;

        let currentIndex = 0;

        let speechRate = 1;

        let paragraphs = [];

        

        // 初始化

        document.addEventListener('DOMContentLoaded', () => {

            paragraphs = Array.from(document.querySelectorAll('[data-speak]'));

            // 默认展开前两个

            document.querySelectorAll('.collapse-content').forEach((el, i) => {

                if (i < 2) el.classList.add('open');

            });

        });

        

        // 切换字体大小

        function toggleTextSize() {

            document.body.classList.toggle('large-text');

            const btn = document.getElementById('size-text');

            btn.textContent = document.body.classList.contains('large-text') ? '标准' : '大字';

        }

        

        // 切换折叠

        function toggleCollapse(btn) {

            const content = btn.closest('.card').querySelector('.collapse-content');

            const arrow = btn.querySelector('svg');

            content.classList.toggle('open');

            if (content.classList.contains('open')) {

                arrow.style.transform = 'rotate(180deg)';

            } else {

                arrow.style.transform = 'rotate(0deg)';

            }

        }

        

        // 设置语速

        function setRate(rate) {

            speechRate = rate;

            // 更新按钮样式

            event.target.parentElement.querySelectorAll('button').forEach(btn => {

                btn.className = 'px-2 py-1 bg-white rounded text-xs border border-gray-200';

            });

            event.target.className = 'px-2 py-1 bg-blue-500 text-white rounded text-xs';

            

            if (isPlaying) {

                stopSpeech();

                startSpeech();

            }

        }

        

        // 播放控制

        function toggleSpeech() {

            if (isPlaying) {

                stopSpeech();

            } else {

                startSpeech();

            }

        }

        

        function startSpeech() {

            if (paragraphs.length === 0) return;

            

            isPlaying = true;

            updatePlayButton();

            

            // 展开所有折叠内容以便朗读

            document.querySelectorAll('.collapse-content').forEach(el => el.classList.add('open'));

            

            speakCurrent();

        }

        

        function speakCurrent() {

            if (currentIndex >= paragraphs.length) {

                stopSpeech();

                currentIndex = 0;

                return;

            }

            

            // 高亮当前段落

            paragraphs.forEach((p, i) => {

                if (i === currentIndex) {

                    p.classList.add('highlight');

                    p.scrollIntoView({ behavior: 'smooth', block: 'center' });

                } else {

                    p.classList.remove('highlight');

                }

            });

            

            const text = paragraphs[currentIndex].innerText;

            utterance = new SpeechSynthesisUtterance(text);

            utterance.lang = 'zh-CN';

            utterance.rate = speechRate;

            utterance.pitch = 1;

            

            utterance.onend = () => {

                currentIndex++;

                speakCurrent();

            };

            

            utterance.onerror = () => {

                stopSpeech();

            };

            

            synth.speak(utterance);

        }

        

        function stopSpeech() {

            isPlaying = false;

            synth.cancel();

            updatePlayButton();

            paragraphs.forEach(p => p.classList.remove('highlight'));

        }

        

        function updatePlayButton() {

            const icon = document.getElementById('btnIcon');

            const text = document.getElementById('btnText');

            const playBtn = document.getElementById('playBtn');

            

            if (isPlaying) {

                icon.textContent = '⏸️';

                text.textContent = '暂停';

                playBtn.innerHTML = '⏸️';

                playBtn.classList.add('speaking');

            } else {

                icon.textContent = '▶️';

                text.textContent = '听全文';

                playBtn.innerHTML = '▶️';

                playBtn.classList.remove('speaking');

            }

        }

        

        // 分享提示

        function shareTips() {

            const ua = navigator.userAgent.toLowerCase();

            if (ua.match(/MicroMessenger/i) == "micromessenger") {

                document.getElementById('wxTips').classList.add('show');

            } else {

                // 非微信环境提示复制链接

                if (navigator.share) {

                    navigator.share({

                        title: '体检真的危险吗？给爸妈的辟谣指南',

                        text: '揭秘那些吓唬老人的健康谣言',

                        url: window.location.href

                    });

                } else {

                    alert('请复制链接发送给亲友，或在微信中打开后转发');

                }

            }

        }

        

        // 页面可见性变化时暂停

        document.addEventListener('visibilitychange', () => {

            if (document.hidden && isPlaying) {

                stopSpeech();

            }

        });

    </script>
